/**
 * Modular PET Bottle Opener - angled / offset tooth version
 *
 * This keeps the parametric Replicad approach, but splits the opener into the
 * same layered structure as the STEP/STL reference:
 * - 60 teeth, 6 degree pitch
 * - every 3rd tooth reaches the cap as a contact fin
 * - the two teeth between are support fins, set back about 0.7 mm
 * - tooth roots are swept/angled about 7.36 degrees from the inner tips
 * - support teeth are slightly phase-offset, matching the STEP tooth pattern
 * - teeth occupy the lower fin band, with a smooth upper bore/lip above them
 */
import type { Drawing, Shape3D } from 'replicad';
import { draw, drawCircle, drawPolysides } from 'replicad';

export const defaultParams = {
  thickness: 10, // Use ~22 for the thick STEP-style body, 8-12 for a flatter print.

  // --- Main opener head, tuned from the uploaded STEP ---
  capDiameter: 29.5, // Contact fin inner diameter: 14.75 mm radius
  finOuterRadius: 25, // Tooth root radius in the STEP
  outerRadius: 27.5, // 55 mm outside envelope from the STL release files
  outerSides: 16, // The STEP outer wall is faceted; use 64 for round
  finHeight: 8, // Height of the working fin band in both STL release variants
  smoothBoreRadius: 24, // Clear bore above the recessed fin band

  finCount: 60, // 6 degree tooth pitch
  contactEvery: 3, // 1 long contact fin, then 2 support fins
  supportSetback: 0.7, // 15.45 mm support radius when cap radius is 14.75

  // Tip/root widths measured from the STEP tooth edges.
  finInnerWidth: 0.86,
  finOuterWidth: 1.8,
  supportFinInnerWidth: 0.855,
  supportFinOuterWidth: 1.8,

  // STEP-style tooth sweep and phase.
  // Negative means the root is clockwise from the inner tip.
  finRootOffsetDeg: -7.36,
  finPhaseDeg: 1.67,
  supportPhaseOffsetDeg: -1.03,

  // --- Optional lower module ---
  // false/false gives the ring-only opener like the blue STEP render.
  addHandle: false,
  secondOpener: false,

  centerDistance: 48,
  neckWidth: 16,

  // Finger/hanging hole, used when addHandle = true and secondOpener = false.
  handleHoleDiameter: 25,
  handleOuterRadius: 17.5,

  // --- Second opener head, used when secondOpener = true ---
  secondCapDiameter: 19,
  secondFinOuterRadius: 16.5,
  secondOuterRadius: 18.5,
  secondFinCount: 36,
  secondFinInnerWidth: 0.55,
  secondFinOuterWidth: 1.05,
  secondSupportFinInnerWidth: 0.65,
  secondSupportFinOuterWidth: 1.05,
  secondContactEvery: 2,
  secondSupportSetback: 0.5,
  secondFinRootOffsetDeg: -7.36,
  secondFinPhaseDeg: 1.67,
  secondSupportPhaseOffsetDeg: -1.0,

  // Chamfer the final solid to avoid the raw flat-extruded look. Keep tiny:
  // larger values touch too many tooth edges and can fail in OCCT.
  edgeChamfer: 0.1,
};

// Small overlap between fin roots and the outer rim for reliable 2D fusing.
const ROOT_OVERLAP = 0.35;
// Overlap between the neck and each round module so they merge cleanly.
const NECK_OVERLAP = 1.8;
const DEG = Math.PI / 180;

type OpenerHead = {
  capDiameter: number;
  finOuterRadius: number;
  outerRadius: number;
  finCount: number;
  finInnerWidth: number;
  finOuterWidth: number;
  supportFinInnerWidth?: number;
  supportFinOuterWidth?: number;
  contactEvery: number;
  supportSetback: number;
  outerSides: number;
  finRootOffsetDeg?: number;
  finPhaseDeg?: number;
  supportPhaseOffsetDeg?: number;
};

type LayeredOpenerHead = OpenerHead & {
  finHeight: number;
  smoothBoreRadius: number;
  thickness: number;
};

/** Point at `radius` and `angle` (radians) around a center. */
function polar(
  center: [number, number],
  radius: number,
  angle: number,
): [number, number] {
  return [
    center[0] + radius * Math.cos(angle),
    center[1] + radius * Math.sin(angle),
  ];
}

/** Closed polygon from a list of 2D points. */
function polygon(points: Array<[number, number]>): Drawing {
  const [first, ...rest] = points;

  if (!first) {
    throw new Error('polygon needs at least one point');
  }

  let pen = draw(first);

  for (const point of rest) {
    pen = pen.lineTo(point);
  }

  return pen.close();
}

/** Ring centered at the origin. Faceted when `sides` is between 3 and 63. */
function annulus(
  outerRadius: number,
  innerRadius: number,
  sides: number,
): Drawing {
  const outer =
    sides >= 3 && sides < 64
      ? drawPolysides(outerRadius, sides)
      : drawCircle(outerRadius);

  return outer.cut(drawCircle(innerRadius));
}

/**
 * A single tapered fin/tooth whose root is angularly offset from its tip.
 *
 * `innerAngle` is the center angle at the cap side.
 * `rootAngleOffset` sweeps the root relative to the inner tip.
 */
function angledRadialFin(
  innerRadius: number,
  outerRadius: number,
  innerWidth: number,
  outerWidth: number,
  innerAngle: number,
  rootAngleOffset: number,
): Drawing {
  const center: [number, number] = [0, 0];

  // Convert tangential widths in mm into angular half-widths.
  const innerHalf = innerWidth / 2 / Math.max(innerRadius, 0.01);
  const outerHalf = outerWidth / 2 / Math.max(outerRadius, 0.01);

  const finOuter = outerRadius + ROOT_OVERLAP;
  const outerAngle = innerAngle + rootAngleOffset;

  return polygon([
    polar(center, innerRadius, innerAngle - innerHalf),
    polar(center, finOuter, outerAngle - outerHalf),
    polar(center, finOuter, outerAngle + outerHalf),
    polar(center, innerRadius, innerAngle + innerHalf),
  ]);
}

/** Builds one opener head as a 2D drawing centered at the origin. */
function buildOpenerHead(head: OpenerHead): Drawing {
  const capRadius = head.capDiameter / 2;
  const ringInner = head.finOuterRadius - ROOT_OVERLAP;

  const contactEvery = Math.max(1, Math.round(head.contactEvery));
  const supportSetback = Math.max(0, head.supportSetback);

  const supportFinInnerWidth =
    head.supportFinInnerWidth ?? head.finInnerWidth;

  const supportFinOuterWidth =
    head.supportFinOuterWidth ?? head.finOuterWidth;

  const finRootOffset = (head.finRootOffsetDeg ?? 0) * DEG;
  const finPhase = (head.finPhaseDeg ?? 0) * DEG;
  const supportPhaseOffset = (head.supportPhaseOffsetDeg ?? 0) * DEG;

  if (capRadius <= 0) {
    throw new Error('capDiameter must be positive');
  }

  if (head.finOuterRadius <= capRadius) {
    throw new Error('finOuterRadius must be larger than capDiameter / 2');
  }

  if (head.outerRadius <= head.finOuterRadius) {
    throw new Error('outerRadius must be larger than finOuterRadius');
  }

  if (head.finCount < 3) {
    throw new Error('finCount must be at least 3');
  }

  if (
    contactEvery > 1 &&
    capRadius + supportSetback >= head.finOuterRadius - 0.25
  ) {
    throw new Error(
      'supportSetback is too large; support fins would not reach the rim cleanly',
    );
  }

  let shape = annulus(head.outerRadius, ringInner, head.outerSides);

  for (let i = 0; i < head.finCount; i++) {
    const isContact = contactEvery <= 1 || i % contactEvery === 0;

    const finInnerRadius = isContact
      ? capRadius
      : capRadius + supportSetback;

    const innerWidth = isContact
      ? head.finInnerWidth
      : supportFinInnerWidth;

    const outerWidth = isContact
      ? head.finOuterWidth
      : supportFinOuterWidth;

    const phaseOffset = isContact ? 0 : supportPhaseOffset;
    const innerAngle =
      finPhase + phaseOffset + (i * 2 * Math.PI) / head.finCount;

    shape = shape.fuse(
      angledRadialFin(
        finInnerRadius,
        head.finOuterRadius,
        innerWidth,
        outerWidth,
        innerAngle,
        finRootOffset,
      ),
    );
  }

  return shape;
}

function extrudeDrawing(drawing: Drawing, height: number, z = 0): Shape3D {
  const solid = drawing.sketchOnPlane().extrude(height);

  if (z === 0) {
    return solid;
  }

  return solid.translate(0, 0, z);
}

/** Builds the reference-style head: recessed fin band + smooth upper bore. */
function buildLayeredOpenerHead(head: LayeredOpenerHead): Shape3D {
  const finHeight = Math.min(Math.max(head.finHeight, 0.1), head.thickness);
  const upperHeight = Math.max(head.thickness - finHeight, 0);

  let solid = extrudeDrawing(buildOpenerHead(head), finHeight);

  if (upperHeight > 0) {
    if (head.smoothBoreRadius <= 0) {
      throw new Error('smoothBoreRadius must be positive');
    }

    if (head.smoothBoreRadius >= head.outerRadius) {
      throw new Error('smoothBoreRadius must be smaller than outerRadius');
    }

    solid = solid.fuse(
      extrudeDrawing(
        annulus(head.outerRadius, head.smoothBoreRadius, head.outerSides),
        upperHeight,
        finHeight,
      ),
    );
  }

  return solid;
}

/**
 * Rectangular neck that overlaps the two round modules but stops short of
 * their centers so it never fills a cap opening. Returns null when the modules
 * already overlap enough that no neck is needed.
 */
function buildBridge(
  centerA: [number, number],
  centerB: [number, number],
  radiusA: number,
  radiusB: number,
  width: number,
): Drawing | null {
  const vx = centerB[0] - centerA[0];
  const vy = centerB[1] - centerA[1];
  const length = Math.hypot(vx, vy);

  if (length < 1e-6) {
    return null;
  }

  const ux = vx / length;
  const uy = vy / length;
  const nx = -uy;
  const ny = ux;

  const startDist = Math.max(radiusA - NECK_OVERLAP, 0);
  const endDist = Math.max(radiusB - NECK_OVERLAP, 0);

  const sx = centerA[0] + ux * startDist;
  const sy = centerA[1] + uy * startDist;
  const ex = centerB[0] - ux * endDist;
  const ey = centerB[1] - uy * endDist;

  // Modules overlap enough that no neck is needed.
  if ((ex - sx) * ux + (ey - sy) * uy <= 0) {
    return null;
  }

  const hw = width / 2;

  return polygon([
    [sx + nx * hw, sy + ny * hw],
    [ex + nx * hw, ey + ny * hw],
    [ex - nx * hw, ey - ny * hw],
    [sx - nx * hw, sy - ny * hw],
  ]);
}

export default function main(p = defaultParams): Shape3D {
  const topCenter: [number, number] = [0, 0];
  const lowerCenter: [number, number] = [0, -p.centerDistance];

  let body = buildLayeredOpenerHead({
    capDiameter: p.capDiameter,
    finOuterRadius: p.finOuterRadius,
    outerRadius: p.outerRadius,
    finCount: p.finCount,
    finInnerWidth: p.finInnerWidth,
    finOuterWidth: p.finOuterWidth,
    supportFinInnerWidth: p.supportFinInnerWidth,
    supportFinOuterWidth: p.supportFinOuterWidth,
    contactEvery: p.contactEvery,
    supportSetback: p.supportSetback,
    outerSides: p.outerSides,
    finRootOffsetDeg: p.finRootOffsetDeg,
    finPhaseDeg: p.finPhaseDeg,
    supportPhaseOffsetDeg: p.supportPhaseOffsetDeg,
    finHeight: p.finHeight,
    smoothBoreRadius: p.smoothBoreRadius,
    thickness: p.thickness,
  });

  let lowerOuterRadius = 0;
  let lower: Shape3D | null = null;
  let handleHole: Drawing | null = null;

  if (p.secondOpener) {
    lowerOuterRadius = p.secondOuterRadius;

    lower = buildLayeredOpenerHead({
      capDiameter: p.secondCapDiameter,
      finOuterRadius: p.secondFinOuterRadius,
      outerRadius: p.secondOuterRadius,
      finCount: p.secondFinCount,
      finInnerWidth: p.secondFinInnerWidth,
      finOuterWidth: p.secondFinOuterWidth,
      supportFinInnerWidth: p.secondSupportFinInnerWidth,
      supportFinOuterWidth: p.secondSupportFinOuterWidth,
      contactEvery: p.secondContactEvery,
      supportSetback: p.secondSupportSetback,
      outerSides: 64,
      finRootOffsetDeg: p.secondFinRootOffsetDeg,
      finPhaseDeg: p.secondFinPhaseDeg,
      supportPhaseOffsetDeg: p.secondSupportPhaseOffsetDeg,
      finHeight: Math.min(p.finHeight, p.thickness),
      smoothBoreRadius: Math.min(
        p.secondFinOuterRadius - 0.5,
        p.secondOuterRadius - 1,
      ),
      thickness: p.thickness,
    }).translate(lowerCenter[0], lowerCenter[1], 0);
  } else if (p.addHandle) {
    lowerOuterRadius = p.handleOuterRadius;

    lower = extrudeDrawing(
      drawCircle(p.handleOuterRadius).translate(lowerCenter[0], lowerCenter[1]),
      p.thickness,
    );

    handleHole = drawCircle(p.handleHoleDiameter / 2).translate(
      lowerCenter[0],
      lowerCenter[1],
    );
  }

  if (lower) {
    const bridge = buildBridge(
      topCenter,
      lowerCenter,
      p.outerRadius,
      lowerOuterRadius,
      p.neckWidth,
    );

    if (bridge) {
      body = body.fuse(extrudeDrawing(bridge, p.thickness));
    }

    body = body.fuse(lower);

    if (handleHole) {
      body = body.cut(extrudeDrawing(handleHole, p.thickness));
    }
  }

  // The dual-head bridge introduces tight intersections that OCCT cannot
  // reliably chamfer, so keep that variant sharp rather than failing export.
  if (p.edgeChamfer > 0 && !p.secondOpener) {
    return body.chamfer(p.edgeChamfer);
  }

  return body;
}
